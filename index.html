Método de la burbuja

Iteraciones: 0

Función que ordena un arreglo de 6 enteros con el método de la burbuja


            function metodoBurbuja(arr) {
                let n = arr.length;
                let array = [10, 20, 30, 40, 50, 60];
              
                for (let i = 0; i < n - 1; i++) {
                  for (let j = 0; j < n - 1 - i; j++) {
                    if (arr[j] > arr[j + 1]) {
                      let temp = arr[j];
                      arr[j] = arr[j + 1];
                      arr[j + 1] = temp;
                    }
                  }
                }
              
                return arr;
              }
              
              console.log("Arreglo original:", array);
              
              let arregloOrdenado = metodoBurbuja(array);
              console.log("Array ordenado:", arregloOrdenado);
